int Area;
int Radius=0;

void setup()
{
 Serial.begin(9600); 
}

void loop()
{
  Serial.println("Enter the Radius : ");
  while(Serial.available()==0){}
  Radius=Serial.parseInt();
  Serial.print(Radius);
  Serial.println("\n");
  Area=3.14*Radius*Radius;
  Serial.println("The Area of the Circle is : ");
  Serial.println(Area);
  Serial.println("\n");
 }
